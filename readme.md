# commands to run when starting 

# kill all processes
killall -9 node

# initialize node.js app
npm init

# server package
npm install express

# restart automatically the server after saving changes in the code (update script to start: "nodemon app.js")
npm install nodemon

# install mongoose to connect to mongoDb
npm install mongoose

# install dotenv to use .env config
npm install dotenv

# install body-parser to parse the data as json  
npm install body-parser

# install joi for input validation  
npm install joi

# install bcryptjs for encrypt/decrypt password during user registration and login validation  
npm install bcryptjs

# install jsonwebtoken for session validation, generating an AUTH-TOKEN
npm install jsonwebtoken

# start Server (type in terminal)
npm start
